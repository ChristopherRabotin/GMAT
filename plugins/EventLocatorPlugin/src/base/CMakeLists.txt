MESSAGE("==============================")
MESSAGE("GMAT EventLocator Plugin setup " ${VERSION}) 

if(WIN32)
	SET(PLUGINTARGET  "libEventLocator")
else()
	SET(PLUGINTARGET "EventLocator")
endif()

SET(TargetName ${PLUGINTARGET})

# Add Preproccessor Definitions
ADD_DEFINITIONS("-DLOCATOR_EXPORTS")

# ====================================================================
# list of directories containing source/header files
SET(PLUGIN_DIRS event factory include locator plugin)

# ====================================================================
# source files
SET(PLUGIN_SRCS
    event/Antumbra.cpp
    event/Contact.cpp
    event/Eclipse.cpp
    event/Elevation.cpp
    event/LineOfSight.cpp
    event/Penumbra.cpp
    event/Umbra.cpp
    factory/EventLocatorFactory.cpp
    locator/ContactLocator.cpp
    locator/EclipseLocator.cpp
    plugin/GmatPluginFunctions.cpp
)

# ====================================================================
# libraries
if(WIN32)
	LINK_LIBRARIES(libGmatBase)
else()
	LINK_LIBRARIES(GmatBase)
endif()

# ====================================================================
# Setup common plugin definitions, targets, etc.
_SETUPPLUGIN(${TargetName} "${PLUGIN_DIRS}" "${PLUGIN_SRCS}")

INSTALL( TARGETS ${TargetName}
  DESTINATION plugins
  )
